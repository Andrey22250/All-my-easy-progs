#define _CRT_SECURE_NO_WARNINGS
#include <iostream>

int main()
{
	setlocale(LC_ALL, "rus"); //Установка русского языка
	int e, b, c, a;
	printf("Введите трехзначное число: ");	//Ввод числа
	scanf("%d", &e);
	b = e % 10 + e / 10 % 10 + e / 100; //Сумма цифр
	if (b >= 10 && b <= 99)
		printf("\nСумма цифр - двузначное число");
	else
		printf("\nСумма цифр - не двузначное число");
	c = (e % 10) * (e / 10 % 10) * (e / 100);	//Произведение цифр
	if (c >= 100 && c <= 999)
		printf("\nПроизведение цифр - трёхзначное число\n");
	else
		printf("\nПроизведение цифр - не трёхзначное число\n");
	printf("\nВведите a: ");	//Ввод числа а
	scanf("%d", &a);
	if (c > a)	//Сравнение произведения цифр и а
		printf("\nПроизведение цифр больше а");
	else if (c == a)
		printf("\nПроизведение цифр равно а");
	else
		printf("\nПроизведение цифр меньше а");
	if (b % 5 == 0) //Кратна ли сумма цифр пяти
		printf("\nСумма цифр кратна 5");
	else
		printf("\nСумма цифр не кратна 5");
	if (b % a == 0)	//Кратна ли сумма цифр числу а
		printf("\nСумма цифр кратна a\n");
	else
		printf("\nСумма цифр не кратна a\n");
}
